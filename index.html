To enable your generated feed.xml, include it as a link in your layout:

<!-- src/routes/+layout.svelte --> <script lang="ts"> import {set_blog_feed, get_blog_feed} from '@ryanatkn/fuz_blog/blog.js'; import {feed} from '$routes/blog/feed.js'; set_blog_feed(feed); // To get the feed from any descendent: const feed = get_blog_feed(); </script> <svelte:head> <title>(your title here)</title> <link rel="alternate" type="application/atom+xml" title="Atom" href={feed.atom.feed_url} /> </svelte:head>

$ gro post "Some post title" $ gro post --help
